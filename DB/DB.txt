drop database dbproject;
create database dbProject;
 use dbProject;
create table Department(
	DeptID char(1),
	DeptName varchar(15),
	primary key(DeptID)
	);
create table Account(
	Username varchar(15),
	type varchar(15),
	primary key(Username)
	);
create table Employee(
	EmpId char(9),
	FName varchar(15),
	LName varchar(15),
	email varchar(30),
	phone char(10),
	DeptID char(1),
	Username varchar(15),
	primary key(EmpId),
	foreign key(DeptID) references Department(DeptID),
	foreign key(Username) references Account(Username)
	);
create table Administrator(
	AdminID char(9),
	FName varchar(15),
	LName varchar(15),
	email varchar(30),
	phone char(10),
	Username varchar(15),
	primary key(AdminID),
	foreign key(Username) references Account(Username)
	);
create table Client(
	ClientId char(9),
	FName varchar(15),
	LName varchar(15),
	email varchar(30),
	phone char(15),
	Username varchar(15),
	primary key(ClientId),
	foreign key(Username) references Account(Username)
	);
create table Incident(
	IncidentId int(10) UNSIGNED NOT NULL AUTO_INCREMENT,
	type varchar(15),
	status varchar(15),
	urgency int,
	impact int,
	description varchar(144),
	ClientId char(9),
	AdminID char(9),
	DateSubmitted date,
	primary key(IncidentId),
	foreign key(ClientId) references Client(ClientId),
	foreign key(AdminID) references Administrator(AdminID)
	);
create table IncidentHistory(
	IncidentId int(10) UNSIGNED NOT NULL,
	EmpId char(9),
	timesWorked int(10) UNSIGNED NOT NULL AUTO_INCREMENT,
	TimeStart time(3),
	TimeEnd time(3),
	DateWorked date,
	description varchar(144),
	primary key(IncidentId, EmpId, timesWorked),
	foreign key(IncidentId) references Incident(IncidentId),
	foreign key(EmpId) references Employee(EmpId)
	)ENGINE=MyISAM;

create view IncidentSummary as
select I.IncidentId, I.description, count(H.EmpID) as peopleWorking, 
    sum(time_to_sec(subtime(H.TimeEnd, H.TimeStart))/60) as minutesWorked, 
    sum(time_to_sec(subtime(H.TimeEnd, H.TimeStart))/3600) as hoursWorked
from Incident I, IncidentHistory H
where I.IncidentId = H.IncidentId
group by I.IncidentId;

create view ej1 as
SELECT B.EmpId, A.FName, A.LName
FROM employee A, incidenthistory B
WHERE A.EmpId = B.EmpId
GROUP BY B.EmpId
HAVING COUNT(A.EmpId) >= ALL(
        SELECT COUNT(EmpId)
        FROM incidenthistory
        GROUP BY EmpId
        );

create view ej5 as
SELECT B.DeptID, A.*
FROM incident A, employee B, incidenthistory C
WHERE A.IncidentId = C.IncidentId AND B.EmpId = C.EmpId
ORDER BY B.DeptID;

create view ej6 as
SELECT D.DeptID, D.DeptName
FROM incident A, employee B, incidenthistory C, department D
WHERE A.IncidentId = C.IncidentId AND B.EmpId = C.EmpId AND B.DeptID = D.DeptID AND MONTH(A.DateClosed) = MONTH(NOW())
GROUP BY D.DeptId
HAVING COUNT(D.DeptId) >= ALL(
        SELECT COUNT(D1.DeptId)
        FROM incident A1, employee B1, incidenthistory C1, department D1
        WHERE A1.IncidentId = C1.IncidentId AND B1.EmpId = C1.EmpId AND B1.DeptID = D1.DeptID
        GROUP BY D1.DeptId
        );
